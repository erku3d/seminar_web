\begin{frame}[fragile=singleslide]{Was macht Rust sicher?}
	\begin{itemize}
	    \item mutabl borrow  
	    \begin{itemize}
	        \item nur eine Referenz auf eine Ressource und diese hat nur einen Besitzer
	    \end{itemize}
	\end{itemize}
\begin{minipage}[t]{.49\textwidth}
\begin{lstlisting}
fn foo() {	
	let y = vec![1, 2, 3, 4, 5];
	let mut v: Vec<i32> 
		= Vec::new();
	bar(&y, &mut v); // Ok
	bar(&y, &mut y); 
		// Compiler Error
}
\end{lstlisting}				
\end{minipage}
\hfill
\begin{minipage}[t]{.49\textwidth}
\begin{lstlisting}
fn bar(x: &Vec<i32>, 
		y: &mut Vec<i32>){
	for v in x{
		y.push(*v);
	}
}
\end{lstlisting}				
\end{minipage}
\end{frame}
